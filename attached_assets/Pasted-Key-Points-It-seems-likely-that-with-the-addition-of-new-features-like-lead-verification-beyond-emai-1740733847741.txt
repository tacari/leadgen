@login_required
def checkout(package):
    package_map = {p['name'].lower().replace(' ', '-'): p for p in packages}
    if package not in package_map:
        flash('Invalid package')
        return redirect(url_for('pricing'))
    price_ids = {
        'lead-starter': 'price_starter_123',  # Replace with real Stripe Price IDs
        'lead-growth': 'price_growth_456',
        'lead-pro': 'price_pro_789',
        'lead-enterprise': 'price_enterprise_101'
    }
    mode = 'payment' if package == 'lead-starter' else 'subscription'
    session = stripe.checkout.Session.create(
        payment_method_types=['card'],
        line_items=[{'price': price_ids[package], 'quantity': 1}],
        mode=mode,
        success_url=url_for('success', _external=True) + '?session_id={CHECKOUT_SESSION_ID}',
        cancel_url=url_for('pricing', _external=True),
        metadata={'package': package}
    )
    return redirect(session.url)

<script>
    const canvas = document.getElementById('waveCanvas');
    const ctx = canvas.getContext('2d');
    canvas.width = window.innerWidth;
    canvas.height = 300;

    let time = 0;
    function drawWave() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ctx.beginPath();
        for (let x = 0; x < canvas.width; x++) {
            const y = 150 + 20 * Math.sin((x + time) * 0.01);
            ctx.lineTo(x, y);
        }
        ctx.strokeStyle = 'rgba(123, 0, 255, 0.3)';
        ctx.lineWidth = 2;
        ctx.stroke();
        time += 1;
        requestAnimationFrame(drawWave);
    }
    drawWave();
</script>
